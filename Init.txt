## Para Tmnux
# Verificar si TPM está instalado
ls ~/.tmux/plugins/tpm

# Si no está instalado, clona el repositorio de TPM
git clone https://github.com/tmux-plugins/tpm ~/.tmux/plugins/tpm

# Abre Tmux y presiona prefix + I para instalar los plugins
tmux
# Dentro de Tmux, presiona Ctrl + b seguido de I
